Upstox

Upstox

API client class. You need to initialise a single instance of this class per api_key. This module provides an easy to use wrapper over the HTTP APIs. The HTTP calls have been converted to methods and their JSON responses.


var Upstox = require("upstox");
var up = new Upstox("your_apikey", "your_apiSecretKey");
up.getSessionToken({
 "username": "your_clientId", 
 "password": "password"
})
.then(function(response) {
        start();
    })
 .catch(function(err) {
        logger.debug(err.response);
    })

start(){
       up.getProfile()
          .then(function(response){
               //get the result here
        })
        .catch(function(error){
               //catch the error here
         });
}
