  9 /**
 10  * @classdesc API client class. In production, you may initialise a single instance of this class per `api_key`.
 11  * This module provides an easy to use wrapper over the HTTP APIs.
 12  * The HTTP calls have been converted to methods and their JSON responses.
 13  *
 14  * var Upstox = require("Upstox");
 15  *
 16  * var up = new Upstox("your_apikey", "your_apiSecretKey");
 17  *
 18  * up.getSessionToken("clientId", "password")
 19  * 	.then(function(response) {
 20  * 		init();
 21  * 	})
 22  * 	.catch(function(err) {
 23  * 		console.log(err.response);
 24  * 	})
 25  *
 26  * @example <caption>Initialize Upstox object</caption>
 27  * var Upstox = require("Upstox");
 28  * var up = new Upstox("your_apikey", "your_apiSecretKey");
 29  *
 30  * @param apiKey
 31  * @param apiSecretKey
 32  * @param sessionToken
 33  * @constructor
 34  */
 35 function Upstox(apiKey, apiSecretKey, sessionToken) {
 36 
 37     var auth;
 38     this.option = {
 39         apiKey : apiKey,
 40         apiSecretKey :apiSecretKey
 41     };
 42 
 43     if(sessionToken){
 44         auth.sessionToken = sessionToken
 45     }
 46     this.auth = _.extend(this.option, auth);
 47 
 48 }
 49 
 50 /**
 51  * @method getSessionToken
 52  * @description Method to get the session token required to authenticate user
 53  * @instance Static
 54  * @param username valid client id for UPSTOX
 55  * @param password valid password associated with the users client id
 56  * @return sessionToken
 57  */
 58 
 59 Upstox.prototype.getSessionToken = function (username, password){
 60 
 61     var params = {
 62         username : username,
 63         password : password
 64     };
 65     var response = RestHelper.POST("login", params, this.auth);
 66     return response.headers["set-cookie"];
 67 };
